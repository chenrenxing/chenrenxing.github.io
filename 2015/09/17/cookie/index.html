  • avaScript是运行在客户端的脚本，因此一般是不能够设置Session的，因为Session是运行在服务器端的。
  • 而cookie是运行在客户端的，所以可以用JS来设置cookie.

#设置cookie#

  • document.cookie=”skin=”+skin;

#获取cookie#

    var arr;reg=new RegExp(“(^| )”+name+”=([^;]*)(;|$)”);
    if(arr=document.cookie.match(reg)){ 
    $scope.current_skin=arr[2];      
 return;

     $scope.current_skin = "skin-blue";  
return;

    #换肤#

$(‘#change_skin [data-skin]’).unbind(“click”).click
(function(e) {
e.preventDefault();
var skinName = $(this).data(‘skin’);//点击this获取data-shin data 属性
$(‘body’).removeClass($scope.current_skin);
$(‘body’).addClass(skinName);
$scope.current_skin = skinName;
$scope.$apply(function(){
$scope.current_skin=skinName;
});
document.cookie=”skin=”+ $scope.current_skin;//皮肤写入 cookie
});
