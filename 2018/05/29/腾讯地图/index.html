angulrjs工厂模式

(function (angular) {
"use strict";
angular.module('public.map.factory', [])
    .factory('publicMapFactory', [function () {
        //地图
        return {
            //地图初始化
            initMap: function (id) {
                var markersArray = [];
                var center = new qq.maps.LatLng(39.916527,116.397128);
                var map = new qq.maps.Map(document.getElementById(id),{
                    center: center,
                    zoom: 10
                });
                //获取城市列表接口设置中心点
                var citylocation = new qq.maps.CityService({
                    complete : function(result){
                        map.setCenter(result.detail.latLng);
                    }
                });
                //调用searchLocalCity();方法    根据用户IP查询城市信息。
                citylocation.searchLocalCity();
                //调用地址解析类
                var geocoder = new qq.maps.Geocoder({
                    complete : function(result){
                        map.setCenter(result.detail.location);
                        var marker = new qq.maps.Marker({
                            map:map,
                            position: result.detail.location
                        });
                        markersArray.push(marker);
                    }
                });
                return {
                    currentGeocoder:function(location){
                        if (markersArray.length>0) {
                            for (var i in markersArray) {
                                markersArray[i].setMap(null);//清除标记
                            }
                        }
                        //通过getLocation();方法获取位置信息值
                        geocoder.getLocation(location);
                    }
                }
            }
        };
    }]);
})(angular);

//解析数据marker标记
function markData() {
    var latlngs = [];
    for (var j = 0; j < $scope.mapData.length; j++) {
        latlngs.push(new qq.maps.LatLng($scope.mapData[j].lat, $scope.mapData[j].lng));
    }
    for (var i = 0; i < latlngs.length; i++) {
        iconMark(i, infoWin, latlngs, map, $scope.mapData[i].dictId);
    }
    //标记END-------------------------------
    //四种颜色iconMark。循环list 添加坐标
}

function iconMark(n, infoWin, latlngs, map, dictId) {
        var marker = new qq.maps.Marker({
            position: latlngs[n],
            map: map
        });
        markersArray.push(marker);
        var img;
        if (dictId === 'traceabilityChain01') {
            img = 'dist/img/map/orange-red.png';
        } else if (dictId === 'traceabilityChain02') {
            img = 'dist/img/map/blue.png';
        } else if (dictId === 'traceabilityChain03') {
            img = 'dist/img/map/dark-green.png';
        } else if (dictId === 'traceabilityChain04') {
            img = 'dist/img/map/green.png';
        }
        var icon = new qq.maps.MarkerImage(img);
        marker.setIcon(icon);
        qq.maps.event.addListener(marker, 'click', function () {
            $scope.nodeId = "";
}
