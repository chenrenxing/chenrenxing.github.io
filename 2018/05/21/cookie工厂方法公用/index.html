cookie工厂方法公用

发表于 2018-05-21

方法

(function (angular) {
"use strict";
angular.module('login.cookie.Factory', [])
    .factory('loginCookieFactory', [function () {
        //记录账号
        return {
            setCookie: function (name, value, timeout) {
                //如果需要加时间限制
                if(timeout){
                    var d = new Date();
                    //设置小时
                    d.setDate(d.getDate() + timeout);
                    document.cookie = name + '=' + value + ';expires=' + d;
                    //设置秒数写法
                    date.setTime(date.getTime()+5*1000); //设置date为当前时间5秒
                    document.cookie="companyTenantId="+$scope.companyTenantId+";expires="+date.toGMTString();  //将date赋值给expires
                }else {
                    document.cookie = name + '=' + value;
                }

            },
            getCookie: function (name) {
                var arr = document.cookie.split('; ');
                for (var i = 0; i < arr.length; i++) {
                    var arr2 = arr[i].split('='); //['abc','cba']
                    if (arr2[0] == name) {
                        return arr2[1];
                        console.log(arr2[1]);
                    }
                }
                return '';
            }
        }
    }]);
})(angular);

使用

//写入
loginCookieFactory.setCookie(name, value, timeout);
//读取
loginCookieFactory。getCookie(name);
