iframe.style.width= “100%”;
iframe.style.height= “100%”;
iframe.src=CONTEXT_iframe_PATH+this.pageUrl+”?access_token=”+this.Token;
document.getElementById(“preview-iframe”).appendChild(iframe);
console.log(“路径=>:”+CONTEXT_iframe_PATH+this.pageUrl+”?access_token=”+this.Token);
