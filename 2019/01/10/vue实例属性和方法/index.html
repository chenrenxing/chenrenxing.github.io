vue实例属性和方法

发表于 2019-01-10

1. 属性

vm.$el  //DOM对象 vm.$el.style.color='red';
vm.$data //vm.$data.msg
vm.$options  //获取自定义属性 vm.$options.name  vm.$options.age  vm$options.show(); 
vm.$refs //获取所有添加ref属性的元素 <p ref="world">世界</p>  console.log(vm.$refs.hello);   vm.$refs.world.style.color='blue';
var vm=new Vue({
         el:'#itany',
        data:{
            msg:'welcome'
        },
        name:'tom',
        age:24,
        show:function(){
            console.log('show');
        }
    });

2. 方法

  • vm.$mount() //挂载
  • vm.$destroy() //销毁

  • vm.$nextTick(callback)

    //在DOM更新完成后再执行回调函数，一般在修改数据之后使用该方法，以便获取更新后的DOM。Vue实现响应式并不是数据发生改变之后DOM立即变化，需要按一定的策略进行DOM更新，需要时间！！
vm.$nextTick(function(){
    //DOM更新完成，更新完成后再执行此代码
    console.log(vm.$refs.title.textContent);
});
  • vm.$set(object,key,value) //给对象添加属性的值

  • vm.$delete(object,key) //删除对象属性

    methods:{
    doUpdate(){
        this.user.name='汤姆'
    },
    doAdd(){
        // this.user.age=25; //通过普通方式为对象添加属性时vue无法实时监视到
        // this.$set(this.user,'age',18); //通过vue实例的$set方法为对象添加属性，可以实时监视
        // Vue.set(this.user,'age',22);//
        if(this.user.age){
            this.user.age++;
        }else{
            this.$set(this.user,'age',1);

        }
        // console.log(this.user);
    },
    doDelete(){
        if(this.user.age){
            // delete this.user.age; //无效
            this.$delete(this.user,'age');
            console.log(this.user);
        }
    }
}

  • vm.$watch(data,callback[,options])

    //方式1：使用vue实例提供的$watch()方法
    vm.$watch(‘name’,function(newValue,oldValue){

    console.log('name被修改啦，原值：'+oldValue+'，新值：'+newValue);

    });
    //方式2：使用vue实例提供的watch选项
    data:{

    name:'tom',
age:23,
user:{
    id:1001,
    name:'alice'
}

    },
    watch:{ 

    age:(newValue,oldValue) => {
    console.log('age被修改啦，原值：'+oldValue+'，新值：'+newValue);
},
user:{//对象
    handler:(newValue,oldValue) => {
        console.log('user被修改啦，原值：'+oldValue.name+'，新值：'+newValue.name);
    },
    deep:true //深度监视，当对象中的属性发生变化时也会监视
}
