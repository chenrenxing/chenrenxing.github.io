grunt前端打包

发表于 2016-04-06
  • 公司项目一直都是用的后端打包工具gradle，这段时间项目相对空隙，就把项 目中前端打包重构了，采用grunt插件。
  • 压缩工具grunt-contrib-compress
  • 上传工具grunt-nexus-deployer
  • 最后记录Gruntfile.js

    压缩配置grunt-contrib-compress

     compress: {
  main: {
    options: {
      archive: 'nexus/ocps-web-all.zip'
    //生成压缩包到nexus/ocps-web-all.zip
    },
    files: [{
      expand: true,
      cwd: '',
      src: [zipList],//压缩的文件
      dest:'ocps-web-all/'//生成的文件包名
        //这里dest是ocps-web-all.zip压缩包内的文件包名，也可以不要。
    }]
  }
},

    zipList配置

      //压缩zip排除文件
  var zipList = [
'**/*',//所有文件,下面！排除不需要文件

  ];

发布grunt-nexus-deployer

//发布 nexus插件
nexusDeployer: {
  snapshot: {
    options: {
      groupId: "com.cyberkey",
      artifactId: "ocps-web-all",//前端统一zip包名
      version: "1.10.8.1-SNAPSHOT",//版本
      packaging: 'zip',//zip包
      auth: {
        username:'nexus用户名',
        password:'nexus密码'
      },
      url: 'nexus仓库快照版的URL',
      artifact:"nexus/ocps-web-all.zip"//生成压缩包的路径和包名
    }
  },
  release: {
    options: {
      groupId: "com.cyberkey",
      artifactId: "ocps-web-all",//前端统一zip包名

      },
      url: 'nexus仓库发布版的URL',

    }
  }
},

};
