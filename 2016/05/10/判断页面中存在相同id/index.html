判断页面中存在相同id

发表于 2016-05-10

前言：项目是管理系统，采用的angularjs框架，table页加载，不多说。
新建、修改的时候弹出框from表单内里面采用了一些自定义写法、以及3方插件类的，赋值就会存在id赋值。在一次上线过程中，打开了多个菜单，模态框中的input的id赋值出现了问题。分析后多个菜单里面存在了相同id。

后续：id冲突最好的模式遵守开发规范，用这个方法也是对系统的负担增加。

//工厂方法
(function (angular) {
"use strict";
angular.module('check.id.factory', [])
    .factory('checkIdFactory',[function(){
        function checkid(modalId){
             var inputid=[];
            var inputArrayId=$("#"+modalId).find(':input');
            inputArrayId.each(//使用数组的循环函数 循环这个input数组
                function (){
                    var input =$(this);//循环中的每一个input元素
                    inputid.push(input.attr("id"));//查看循环中的每一个input的id
                }
            );
            for(var i in inputid){
                var checkId=inputid[i];
                if( $("body [id="+checkId+"]").length >= 2 ){
                    return (true);
                }
            }
        }
        return {
            checkAllid: function (id) {
                return checkid(id);
            }
        }
    }])
})(angular);

工厂写法2不跟上组件有关系两种返回写法而已

(function (angular) {
'use strict';
        angular.module('check.id.factory', [])
    .factory('checkIdFactory',[function(){$http) {
        var service = {};//定义一个Object对象'
        var allData={
            fixFileData:{},
            navIconQueue:[],
            cacheNavData:[],//用于缓存button图片数据,转换了数组navIconQueue传出
            autoFileData:{}
        };
        service.getData=function(){
            var urlUnClickIcon={};//导航模板上传的未点击图标
            var urlClickIcon={};//导航模板上传的点击图标
            var antoBtnClickIcon={};//自增长buttn跳转的图标
            for (var i in allData.cacheNavData) {
                var eachFile = allData.cacheNavData[i].data;
                if(eachFile.fileType=="urlUnClickIcon"){
                    urlUnClickIcon[eachFile.navId]=eachFile
                }else if(eachFile.fileType=="urlClickIcon"){
                    urlClickIcon[eachFile.navId]=eachFile
                }else if(eachFile.fileType="btnClickIcon"){
                    antoBtnClickIcon[eachFile.navId]=eachFile
                }
            }
            for (var i in urlUnClickIcon) {
                var urlUnClickIconData = urlUnClickIcon[i];
                allData.navIconQueue.push(urlUnClickIconData);
            }
            for (var i in urlClickIcon) {
                var urlClickIconData = urlClickIcon[i];
                allData.navIconQueue.push(urlClickIconData);
            }
            for (var i in antoBtnClickIcon) {
                var antoBtnClickIconData = antoBtnClickIcon[i];
                allData.navIconQueue.push(antoBtnClickIconData);
            }
            return allData;
        };
        service.clearData=function(){
            allData={
                fixFileData:{},
                cacheNavData:[],//用于缓存button图片数据
                navIconQueue:[],
                autoFileData:{}
            };
        }
        //获取固定的图片file对象
        $rootScope.$on('fixDataLink',function(event,data,positionId){
                data.fileContent.PicCutInfoDto=data.PicCutInfoDto;
            data.fileContent.cut=data.cut;
            data.fileContent.position=data.position;
            allData.fixFileData[positionId]=data.fileContent ;
        });
        $rootScope.$on('autoFileData',function(event,data,positionId){
                data.fileContent.PicCutInfoDto=data.PicCutInfoDto;
            data.fileContent.cut=data.cut;
            data.fileContent.position=data.position;
            allData.autoFileData[positionId]=data.fileContent ;
        });
        //获取底部导航的图片file对象
        $rootScope.$on('navIconQueue',function(event,data,positionId){
            allData.cacheNavData.push({
                    data:data
            });
        });
        return service;//返回这个Object对象
    }]);
})(angular);
