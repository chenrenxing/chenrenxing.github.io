方法

(function (angular) {
"use strict";
angular.module('login.cookie.Factory', [])
    .factory('loginCookieFactory', [function () {
        //记录账号
        return {
            setCookie: function (name, value, timeout) {
                //如果需要加时间限制
                if(timeout){
                    var d = new Date();
                    //设置小时
                    d.setDate(d.getDate() + timeout);
                    document.cookie = name + '=' + value + ';expires=' + d;
                    //设置秒数写法
                    date.setTime(date.getTime()+5*1000); //设置date为当前时间5秒
                    document.cookie="companyTenantId="+$scope.companyTenantId+";expires="+date.toGMTString();  //将date赋值给expires
                }else {
                    document.cookie = name + '=' + value;
                }

            },
            getCookie: function (name) {
                var arr = document.cookie.split('; ');
                for (var i = 0; i < arr.length; i++) {
                    var arr2 = arr[i].split('='); //['abc','cba']
                    if (arr2[0] == name) {
                        return arr2[1];
                        console.log(arr2[1]);
                    }
                }
                return '';
            }
        }
    }]);
})(angular);

 function createrandom() {
           //生成随机数
           var newNum = parseInt(Math.random() * (100 - 10 + 1) + 10);
           var num = repeatNum(newNum); //验证是否重复
           if (num) {//有合格随机数
               return num;//返回当前合格随机数
           }
       }
//判断随机数重复
       function repeatNum(newNum) {
           var repeat = 0;
           if (numAll.length > 0) {//数组有值
               for (var i = 0; i < numAll.length; i++) {//循环全部随机数
                   if (numAll[i] === newNum) {//若有相同随机数，重新生成
                       repeat = 1; //有重复值 为1
                       i = numAll.length;//跳出循环
                       createrandom();//重新生成随机数
                   }
               }
           }
           if (repeat === 0) {//无重复
               numAll.push(newNum);//本次随机数加入总随机数组
               return newNum;//返回当前合格随机数
           }
       }
