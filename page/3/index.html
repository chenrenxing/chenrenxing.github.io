'use strict';
var path = require('path');

var RemTransform = function(option) {
var baseFont = option.baseFont, designWidth = option.designWidth;
//创建media query
function _createMedia() {
    var tmp = '';
    var tpl = '@media only screen and (max-width: <%=screenWidth%>px), only screen and (max-device-width:<%=screenWidth%>px) {\n' +
        'html,body {\n' +
        'font-size:<%=fontSize%>px;\n' +
        '}\n' +
        '}\n';
    var screens = [1080, 960, 800, 720, 640, 600, 540, 480, 414, 400, 375, 360, 320, 240];
    for (var i = 0; i < screens.length; i++) {
        tmp += (tpl.replace(/\<%\=screenWidth%\>/g, screens[i]).replace('<%=fontSize%>', (screens[i] / designWidth)*baseFont));
    }
    return tmp;
}
//获取rem值
function _getRem(px) {
    var reg = new RegExp("[0-9]+([.]{1}[0-9]+){0,1}","g"), rem = px;
    var tmp = px.match(reg);
    for (var i = 0; i < tmp.length; i++) {
        if(Number(tmp[i]) === 0 || px.indexOf( tmp[i] + 'px' ) < 0) {
            continue;//0不做处理,数字后面不是px不做处理
        }
        rem = rem.replace(tmp[i] + 'px', (Number(tmp[i]) / baseFont) + 'rem');
    }
    return rem;
}
//获取px值
function _getPx(rem) {
    var reg = new RegExp("[0-9]+([.][0-9]+)?","g"), px = rem;
    var tmp = rem.match(reg);
    for (var i = 0; i < tmp.length; i++) {
        px = px.replace(tmp[i] + 'rem', (Number(tmp[i]) * baseFont) + 'px');
    }
    return px;
}
//末尾自动添加分号
function _addSemicolon (str) {
    var reg = new RegExp(/([0-9]+)([^;{}])?}/g);
    return str.replace(reg, "$1$2;}");
}
function _filterBorder(str) {
    if(str.substr(-6) === 'border'
        || str.substr(-12) === 'border-width'
        || str.substr(-10) === 'border-top'
        || str.substr(-11) === 'border-left'
        || str.substr(-12) === 'border-right'
        || str.substr(-13) === 'border-bottom'
        || str.substr(-13) === 'border-radius'
    ) {
        return true;
    }
    return false;
}
//将px转换成rem
function changeToRem (input) {
    var after = '', tmp = [],
        reg = new RegExp(":[^:]*px([^;/])*;","g"),
        before = _addSemicolon(input);
    var pxArray = before.match(reg);
    if(!pxArray) { return; }
    for (var i = 0; i < pxArray.length; i++) {
        tmp = before.split(pxArray[i]);
        if(option.border && _filterBorder(tmp[0])) {
            continue;
        }
        after += tmp[0];
        if(option.ie8) {
            after += pxArray[i];
            after += tmp[0].substr(Math.max( tmp[0].lastIndexOf(';'), tmp[0].lastIndexOf('{') ) + 1);
        }
        after += _getRem(pxArray[i], baseFont);
        before = before.replace(tmp[0], '').replace(pxArray[i], '');
    }
    if(option.media) {
        return _createMedia() + after + before;
    } else {
        return after + before;
    }
}
//rem转px
function changeToPx (input) {
    var after = '', tmp = [], reg = new RegExp(":.*rem","g"), before = input;
    var pxArray = before.match(reg);
    if(!pxArray) { return; }
    for (var i = 0; i < pxArray.length; i++) {
        tmp = before.split(pxArray[i]);
        after += tmp[0]  + _getPx(pxArray[i], baseFont);
        before = before.replace(tmp[0], '').replace(pxArray[i], '');
    }
    return after + before;
}
return {
    px2rem: changeToRem,
    rem2px: changeToPx
};

后续：id冲突最好的模式遵守开发规范，用这个方法也是对系统的负担增加。

//工厂方法
(function (angular) {

    .factory('checkIdFactory',[function(){
        function checkid(modalId){
             var inputid=[];
            var inputArrayId=$("#"+modalId).find(':input');
            inputArrayId.each(//使用数组的循环函数 循环这个input数组
                function (){
                    var input =$(this);//循环中的每一个input元素
                    inputid.push(input.attr("id"));//查看循环中的每一个input的id
                }
            );
            for(var i in inputid){
                var checkId=inputid[i];
                if( $("body [id="+checkId+"]").length >= 2 ){
                    return (true);
                }
            }
        }
        return {
            checkAllid: function (id) {
                return checkid(id);
            }
        }
    }])

(function (angular) {
'use strict';
        angular.module('check.id.factory', [])
    .factory('checkIdFactory',[function(){$http) {
        var service = {};//定义一个Object对象'
        var allData={
            fixFileData:{},
            navIconQueue:[],
            cacheNavData:[],//用于缓存button图片数据,转换了数组navIconQueue传出
            autoFileData:{}
        };
        service.getData=function(){
            var urlUnClickIcon={};//导航模板上传的未点击图标
            var urlClickIcon={};//导航模板上传的点击图标
            var antoBtnClickIcon={};//自增长buttn跳转的图标
            for (var i in allData.cacheNavData) {
                var eachFile = allData.cacheNavData[i].data;
                if(eachFile.fileType=="urlUnClickIcon"){
                    urlUnClickIcon[eachFile.navId]=eachFile
                }else if(eachFile.fileType=="urlClickIcon"){
                    urlClickIcon[eachFile.navId]=eachFile
                }else if(eachFile.fileType="btnClickIcon"){
                    antoBtnClickIcon[eachFile.navId]=eachFile
                }
            }
            for (var i in urlUnClickIcon) {
                var urlUnClickIconData = urlUnClickIcon[i];
                allData.navIconQueue.push(urlUnClickIconData);
            }
            for (var i in urlClickIcon) {
                var urlClickIconData = urlClickIcon[i];
                allData.navIconQueue.push(urlClickIconData);
            }
            for (var i in antoBtnClickIcon) {
                var antoBtnClickIconData = antoBtnClickIcon[i];
                allData.navIconQueue.push(antoBtnClickIconData);
            }
            return allData;
        };
        service.clearData=function(){
            allData={
                fixFileData:{},
                cacheNavData:[],//用于缓存button图片数据
                navIconQueue:[],
                autoFileData:{}
            };
        }
        //获取固定的图片file对象
        $rootScope.$on('fixDataLink',function(event,data,positionId){
                data.fileContent.PicCutInfoDto=data.PicCutInfoDto;
            data.fileContent.cut=data.cut;
            data.fileContent.position=data.position;
            allData.fixFileData[positionId]=data.fileContent ;
        });
        $rootScope.$on('autoFileData',function(event,data,positionId){
                data.fileContent.PicCutInfoDto=data.PicCutInfoDto;
            data.fileContent.cut=data.cut;
            data.fileContent.position=data.position;
            allData.autoFileData[positionId]=data.fileContent ;
        });
        //获取底部导航的图片file对象
        $rootScope.$on('navIconQueue',function(event,data,positionId){
            allData.cacheNavData.push({
                    data:data
            });
        });
        return service;//返回这个Object对象
    }]);
})(angular);
